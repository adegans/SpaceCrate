# Job termination
# Offered on SpaceCrate planets
# ------------------------------------------------------------------------------------------
mission "spacecrate_employee_quit_1"
	job
	repeat
	name "Quit working for SpaceCrate - Dangerous"
	description "Steal a shipment of Quafe from the factory and gift it to the Quarg in <destination>. The Quarg might reward you for your gift, but SpaceCrate will definitely hunt you down and terminate your employment."
	cargo "Quafe" 20 60
	to offer
		"combat rating" > 8103
		"spacecrate_employee" > 0
	source
		government "SpaceCrate"
	destination
		government "Quarg"
		near "Tarazed" 1 2
	npc
		government "SpaceCrate"
		personality	nemesis opportunistic heroic entering
		system "Alpha Centauri"
		fleet "spacecrate_missionfleet_2"
	npc
		government "SpaceCrate"
		personality	nemesis opportunistic heroic entering
		system 
			near "Cebalrai" 0 2
		fleet "spacecrate_missionfleet_2"
	npc
		government "SpaceCrate"
		personality	nemesis opportunistic heroic entering
		system 
			near "Dabih" 0 1
		fleet "spacecrate_missionfleet_2" 2
	npc
		government "SpaceCrate"
		personality	nemesis opportunistic heroic entering
		system destination
		fleet "spacecrate_missionfleet_2"
	on complete
		payment 10000 500
		"reputation: SpaceCrate" = 2
		"reputation: Quarg" += 30
		clear "spacecrate_employee"
		dialog "The Quarg look pleasantly surprised with your unexpected gift and give you a 'finders fee' of <payment>."
		dialog "You get a message from SpaceCrate HR expressing their dissapointment and they notify you of the termination of your employment as well as a loss of reputation."
	on fail
		fine 100000
		"reputation: SpaceCrate" = 2
		clear "spacecrate_employee"
		dialog "Looks like the SpaceCrate defense fleet caught up to you and they look none too happy. You get a hefty fine of <fine> and reputation loss for your foolishness."
		dialog "SpaceCrate HR informs you that your employment has been terminated."

	on fail
